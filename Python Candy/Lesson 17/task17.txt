# İstifadəçidən nömrə alaq
n = int(input("Bir nömrə daxil edin: "))

# Fibonacci ardıcıllığını hesablama
a, b = 0, 1
while a <= n:
    print(a, end=" ")
    a, b = b, a + b


# For döngəsindən istifadə edərək vurma cədvəlini çap edin
for i in range(1, 11):
    for j in range(1, 11):
        print(f"{i} * {j} = {i * j}")
    print()  # Hər sətirdən sonra boş sətir əlavə edin


# For döngəsindən istifadə edərək tək nömrələri çap edin
print("Tək ədədlər:")
for num in range(1, 101, 2):
    print(num, end=" ")


# While döngəsindən istifadə edərək cüt nömrələri çap edin
print("\n\nCüt ədədlər:")
num = 2
while num <= 100:
    print(num, end=" ")
    num += 2
